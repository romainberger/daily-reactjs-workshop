Why we should use it

  • Not a complete MVC framework
  • Reusable components
  • Fast and easy
  • Used in production apps
  • Well documented
  • Easily testable

Tests (for real)

  • Test utilities available
  • Jest

/** @jsx React.DOM */

jest.dontMock('hello')

var React = require('react/addons'),
  TestUtils = React.addons.TestUtils,
  Hello = require('hello')

describe('Hello', function() {
  it('Should return a hello', function() {
    var component = TestUtils.renderIntoDocument(
      <Hello name="Romain" />
    )

    var text = TestUtils.findRenderedDOMComponentWithTag(component, 'div')

    expect(text.getDOMNode().textContent()).toEqual('Hello Romain')
  })
})
